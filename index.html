        class Bullet {

            constructor(x, y, angle, speed) {

                this.x = x; this.y = y; this.angle = angle; this.speed = speed;

            }

            update() {

                this.x += Math.cos(this.angle) * this.speed;

                this.y += Math.sin(this.angle) * this.speed;

            }

            draw() {

                ctx.beginPath(); ctx.arc(this.x, this.y, 4, 0, Math.PI * 2);

                ctx.fillStyle = '#f87171'; ctx.shadowBlur = 10; ctx.shadowColor = '#f87171';

                ctx.fill(); ctx.shadowBlur = 0;

            }

        }


        class Enemy {

            constructor(type) {

                this.type = type;

                this.radius = 12;

                this.timer = 0;

                this.lifeTimer = 0;

                const side = Math.floor(Math.random() * 4);

                if(side === 0) { this.x = Math.random() * canvas.width; this.y = -20; }

                else if(side === 1) { this.x = canvas.width + 20; this.y = Math.random() * canvas.height; }

                else if(side === 2) { this.x = Math.random() * canvas.width; this.y = canvas.height + 20; }

                else { this.x = -20; this.y = Math.random() * canvas.height; }

            }

            update(dt) {

                this.timer += dt;

                if(this.type === 'CHASER') {

                    this.lifeTimer += dt;

                    if(this.lifeTimer > 8) return true;

                    const angle = Math.atan2(player.y - this.y, player.x - this.x);

                    this.x += Math.cos(angle) * 1.8;

                    this.y += Math.sin(angle) * 1.8;

                } else {

                    let target = items.find(i => i.type === 'COIN');

                    if(target) {

                        const angle = Math.atan2(target.y - this.y, target.x - this.x);

                        this.x += Math.cos(angle) * 3; this.y += Math.sin(angle) * 3;

                        if(Math.hypot(this.x - target.x, this.y - target.y) < 20) {

                            items.splice(items.indexOf(target), 1);

                            return true;

                        }

                    } else {

                        this.x += Math.cos(this.timer) * 2;

                    }

                }

                return false;

            }

            draw() {

                ctx.fillStyle = this.type === 'CHASER' ? '#f87171' : '#ffffff';

                ctx.beginPath();

                if(this.type === 'CHASER') {

                    ctx.moveTo(this.x, this.y - 12);

                    ctx.lineTo(this.x + 10, this.y + 10);

                    ctx.lineTo(this.x - 10, this.y + 10);

                } else {

                    ctx.rect(this.x - 10, this.y - 10, 20, 20);

                }

                ctx.fill();

            }

        }


        class Item {

            constructor() {

                const angle = Math.random() * Math.PI * 2;

                const r = Math.random() * (safeZone.radius * 0.8);

                this.x = safeZone.x + Math.cos(angle) * r;

                this.y = safeZone.y + Math.sin(angle) * r;

                this.type = Math.random() > 0.3 ? 'COIN' : 'HP';

            }

            draw() {

                ctx.fillStyle = this.type === 'COIN' ? '#fbbf24' : '#22d3ee';

                ctx.shadowBlur = 10; ctx.shadowColor = ctx.fillStyle;

                ctx.beginPath(); ctx.arc(this.x, this.y, 8, 0, Math.PI * 2);

                ctx.fill(); ctx.shadowBlur = 0;

            }

        }


        function startGame() {

            document.getElementById('loginScreen').classList.add('hidden');

            document.getElementById('gameScreen').style.display = 'block';

            init();

            gameActive = true;

            lastTime = performance.now();

            requestAnimationFrame(gameLoop);

        }


        function init() {

            canvas.width = window.innerWidth;

            canvas.height = window.innerHeight;

            safeZone.x = canvas.width / 2;

            safeZone.y = canvas.height / 2;

            safeZone.maxRadius = (Math.min(canvas.width, canvas.height) / 2) - 30;

            safeZone.radius = safeZone.maxRadius;

            player.x = safeZone.x;

            player.y = safeZone.y;

            player.targetX = player.x;

            player.targetY = player.y;

        }


        function gameLoop(time) {

            if(!gameActive) return;

            const dt = (time - lastTime) / 1000;

            lastTime = time;


            ctx.fillStyle = '#050505';

            ctx.fillRect(0, 0, canvas.width, canvas.height);


            safeZone.radius -= (5 + level) * dt;

            if(safeZone.radius < safeZone.minRadius) safeZone.radius = safeZone.minRadius;

            ctx.beginPath(); ctx.arc(safeZone.x, safeZone.y, safeZone.radius, 0, Math.PI * 2);

            ctx.strokeStyle = '#00f2ff'; ctx.lineWidth = 2; ctx.stroke();

            ctx.fillStyle = 'rgba(0, 242, 255, 0.05)'; ctx.fill();


            player.x += (player.targetX - player.x) * 0.12;

            player.y += (player.targetY - player.y) * 0.12;

            if(Math.hypot(player.x - safeZone.x, player.y - safeZone.y) + player.radius > safeZone.radius) {

                health -= 15 * dt;

            }


            ctx.fillStyle = '#ff00ff'; ctx.shadowBlur = 15; ctx.shadowColor = '#ff00ff';

            ctx.beginPath(); ctx.arc(player.x, player.y, player.radius, 0, Math.PI * 2); ctx.fill();

            ctx.shadowBlur = 0;


            bulletTimer += dt;

            if(bulletTimer > 1.2 / (1 + level * 0.1)) {

                const a = Math.random() * Math.PI * 2;

                bullets.push(new Bullet(

                    safeZone.x + Math.cos(a) * canvas.width,

                    safeZone.y + Math.sin(a) * canvas.height,

                    Math.atan2(player.y - (safeZone.y + Math.sin(a) * canvas.height), player.x - (safeZone.x + Math.cos(a) * canvas.width)),

                    2 + level * 0.5

                ));

                bulletTimer = 0;

            }


            enemyTimer += dt;

            if(enemyTimer > 4 / (1 + level * 0.2)) {

                enemies.push(new Enemy(Math.random() > 0.5 ? 'CHASER' : 'THIEF'));

                enemyTimer = 0;

            }


            if(Math.random() < 0.01 && items.length < 5) items.push(new Item());


            bullets.forEach((b, i) => {

                b.update(); b.draw();

                if(Math.hypot(b.x - player.x, b.y - player.y) < player.radius + 5) {

                    health -= 10; bullets.splice(i, 1);

                }

            });


            enemies.forEach((e, i) => {

                if(e.update(dt)) enemies.splice(i, 1);

                else {

                    e.draw();

                    if(Math.hypot(e.x - player.x, e.y - player.y) < player.radius + 12) {

                        health -= 20; enemies.splice(i, 1);

                    }

                }

            });


            items.forEach((it, i) => {

                it.draw();

                if(Math.hypot(it.x - player.x, it.y - player.y) < player.radius + 15) {

                    if(it.type === 'COIN') {

                        score += 500;

                        safeZone.radius = Math.min(safeZone.maxRadius, safeZone.radius + 50);

                    } else {

                        health = Math.min(100, health + 20);

                    }

                    items.splice(i, 1);

                }

            });


            score += 0.1;

            level = Math.floor(score / 2000) + 1;

            document.getElementById('scoreVal').innerText = Math.floor(score);

            document.getElementById('healthVal').innerText = Math.max(0, Math.ceil(health));

            document.getElementById('hpBar').style.width = `${health}%`;

            document.getElementById('levelVal').innerText = level;


            if(health <= 0) endGame(); else requestAnimationFrame(gameLoop);

        }


        async function endGame() {

            gameActive = false;

            document.getElementById('gameOver').classList.remove('hidden');

            document.getElementById('finalScore').innerText = Math.floor(score);

            const statusEl = document.getElementById('saveStatus');

            const res = await apiCall({ action: 'saveScore', username: currentUser, score: Math.floor(score) });

            if(res.status === 'success') {

                statusEl.innerText = res.updated ? "新紀錄已同步！" : "同步完成";

                if(res.updated) document.getElementById('hiScoreVal').innerText = Math.floor(score);

            } else {

                statusEl.innerText = "雲端同步失敗";

            }

        }


        document.getElementById('restartBtn').onclick = () => {

            health = 100; score = 0; level = 1;

            bullets = []; enemies = []; items = [];

            document.getElementById('gameOver').classList.add('hidden');

            startGame();

        };

        window.addEventListener('touchmove', e => { 

            e.preventDefault(); 

            player.targetX = e.touches[0].clientX; 

            player.targetY = e.touches[0].clientY; 

        }, { passive: false });
